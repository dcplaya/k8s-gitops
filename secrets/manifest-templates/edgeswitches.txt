---
apiVersion: networking.k8s.io/v1beta1
kind: Ingress
metadata:
  name: external-drew-es1
  annotations:
    kubernetes.io/ingress.class: nginx
    nginx.ingress.kubernetes.io/backend-protocol: "HTTPS"
spec:
  rules:
  - host: es1.network.$DOMAIN
    http:
      paths:
      - backend:
          serviceName: external-drew-es1
          servicePort: 443
  tls:
  - hosts:
    - es1.network.$DOMAIN
---
apiVersion: v1
kind: Service
metadata:
  name: external-drew-es1
spec:
  ports:
  - name: https
    port: 443
